### 2.2 main.rs 플로우 변경

**변경 대상**: `src/main.rs`

```rust
fn main() {
    let cli = Cli::parse();
    let path = cli.path.clone();

    match git::analyze(&path, &cli.period) {
        Ok(analysis) => {
            let score = scoring::calculate_vibe_score(&analysis);

            if cli.badge {
                // 배지 생성 경로
                let out_path = cli.output.clone()
                    .unwrap_or_else(|| PathBuf::from("vibe-badge.svg"));
                if let Err(e) = output::render_badge(&score, &out_path) {
                    eprintln!("배지 생성 오류: {}", e);
                    process::exit(1);
                }
                println!("배지 생성 완료: {}", out_path.display());
            } else if cli.share {
                // 이미지 생성 경로
                let ext = if cli.format == "png" { "png" } else { "svg" };
                let out_path = cli.output.clone()
                    .unwrap_or_else(|| PathBuf::from(format!("vibe-report.{}", ext)));
                if let Err(e) = output::render_share(&score, &analysis, &out_path, &cli.format) {
                    eprintln!("이미지 생성 오류: {}", e);
                    process::exit(1);
                }
                println!("이미지 생성 완료: {}", out_path.display());
            } else {
                // 기존 터미널/JSON 출력
                if let Err(e) = output::render(&score, &analysis, cli.json, cli.leaderboard) {
                    eprintln!("출력 오류: {}", e);
                    process::exit(1);
                }
            }
        }
        Err(e) => {
            eprintln!("오류: {}", e);
            process::exit(1);
        }
    }
}
```

```rust
use std::path::Path;
use crate::error::VibeError;
use crate::git::AnalysisResult;
use crate::scoring::VibeScore;

// 기존 render() 유지

/// 공유용 이미지 생성
pub fn render_share(
    score: &VibeScore,
    analysis: &AnalysisResult,
    path: &Path,
    format: &str,
) -> Result<(), VibeError> {
    let svg_content = svg::render_svg(score, analysis);

    match format {
        "png" => svg::save_as_png(&svg_content, path),
        _ => {
            std::fs::write(path, &svg_content)
                .map_err(|e| VibeError::Io(e))
        }
    }
}

/// 배지 SVG 생성
pub fn render_badge(
    score: &VibeScore,
    path: &Path,
) -> Result<(), VibeError> {
    let svg_content = badge::render_badge(score);
    std::fs::write(path, &svg_content)
        .map_err(|e| VibeError::Io(e))
}
```

```rust
use crate::git::AnalysisResult;
use crate::scoring::VibeScore;
use crate::error::VibeError;

/// 등급별 테마 색상
fn grade_color(grade: &crate::scoring::grades::VibeGrade) -> &'static str {
    use crate::scoring::grades::VibeGrade;
    match grade {
        VibeGrade::Zen => "#4c1",
        VibeGrade::Chill => "#97CA00",
        VibeGrade::Active => "#dfb317",
        VibeGrade::Tense => "#fe7d37",
        VibeGrade::Stressed => "#e05d44",
        VibeGrade::Chaotic => "#e05d44",
        VibeGrade::Abandoned => "#9f9f9f",
    }
}

/// 점수 바 SVG rect 생성 (0~100 → 0~200px 바)
fn score_bar_svg(label: &str, value: f64, y: u32, bar_color: &str) -> String {
    let bar_width = 200;
    let filled = ((value / 100.0) * bar_width as f64).round() as u32;
    format!(
        r#"  <text x="40" y="{y}" fill="#c0caf5" font-size="14" font-family="system-ui">{label}</text>
  <rect x="220" y="{bar_y}" width="{bar_width}" height="16" rx="4" fill="#2a2b3d"/>
  <rect x="220" y="{bar_y}" width="{filled}" height="16" rx="4" fill="{bar_color}"/>
  <text x="430" y="{y}" fill="#c0caf5" font-size="14" font-family="system-ui">{score}</text>"#,
        y = y,
        bar_y = y - 12,
        bar_width = bar_width,
        filled = filled,
        bar_color = bar_color,
        label = label,
        score = format!("{:.0}", value),
    )
}

/// 전체 SVG 리포트 이미지 생성
pub fn render_svg(score: &VibeScore, analysis: &AnalysisResult) -> String {
    let accent = grade_color(&score.grade);
    let grade_str = score.grade.to_string();

    let bars = [
        ("Commit Frequency", score.commit_frequency),
        ("Hotspot Health", score.hotspot_score),
        ("Ghost Files", score.ghost_score),
        ("Commit Mood", score.mood_score),
        ("Bus Factor", score.bus_factor_score),
        ("Activity Trend", score.trend_score),
    ];

    let mut bar_svgs = String::new();
    for (i, (label, value)) in bars.iter().enumerate() {
        let y = 220 + (i as u32 * 28);
        bar_svgs.push_str(&score_bar_svg(label, *value, y, accent));
        bar_svgs.push('\n');
    }

    // 감정 바: 가로 비율 막대
    let mood = &analysis.mood;
    let mood_segments = [
        ("😊", mood.happy, "#9ece6a"),
        ("😰", mood.stressed, "#e05d44"),
        ("🧹", mood.cleanup, "#7aa2f7"),
        ("💀", mood.scary, "#9f9f9f"),
        ("🤷", mood.other, "#565f89"),
    ];

    let mut mood_bar = String::new();
    let mood_bar_width = 400u32;
    let mut mood_x = 40u32;
    let mood_y = 410u32;
    for (emoji, ratio, color) in &mood_segments {
        let w = (*ratio * mood_bar_width as f64).round() as u32;
        if w > 0 {
            mood_bar.push_str(&format!(
                r#"  <rect x="{}" y="{}" width="{}" height="18" fill="{}"/>
  <text x="{}" y="{}" fill="white" font-size="11" font-family="system-ui">{} {:.0}%</text>
"#,
                mood_x, mood_y, w, color,
                mood_x + 2, mood_y + 14, emoji, ratio * 100.0
            ));
        }
        mood_x += w;
    }

    format!(
        r#"<svg xmlns="http://www.w3.org/2000/svg" width="800" height="480" viewBox="0 0 800 480">
  <rect width="800" height="480" rx="16" fill="#1a1b26"/>

  <!-- Header -->
  <text x="40" y="45" fill="#7aa2f7" font-size="22" font-weight="bold" font-family="system-ui">🎭 Vibe Check: {repo_name}</text>
  <line x1="40" y1="60" x2="760" y2="60" stroke="#7aa2f7" stroke-width="2"/>

  <!-- Grade -->
  <text x="40" y="100" fill="{accent}" font-size="28" font-weight="bold" font-family="system-ui">Overall Vibe: {grade}</text>
  <text x="40" y="130" fill="#565f89" font-size="16" font-family="system-ui">Score: {score:.0}/100</text>

  <!-- Stats -->
  <text x="40" y="165" fill="#c0caf5" font-size="14" font-family="system-ui">📊 Period: {period}  |  Commits: {commits}  |  Files: {files}</text>

  <!-- Separator -->
  <line x1="40" y1="180" x2="760" y2="180" stroke="#2a2b3d" stroke-width="1"/>

  <!-- Score Bars -->
  <text x="40" y="205" fill="#7aa2f7" font-size="16" font-weight="bold" font-family="system-ui">📈 Sub Scores</text>
{bars}
  <!-- Mood -->
  <text x="40" y="{mood_label_y}" fill="#7aa2f7" font-size="16" font-weight="bold" font-family="system-ui">🎭 Commit Mood</text>
{mood_bar}
  <!-- Footer -->
  <line x1="40" y1="445" x2="760" y2="445" stroke="#2a2b3d" stroke-width="1"/>
  <text x="40" y="468" fill="#565f89" font-size="12" font-family="system-ui">Generated by git-vibe · github.com/calintzy/git-vibe</text>
</svg>"#,
        repo_name = xml_escape(&analysis.repo_name),
        accent = accent,
        grade = xml_escape(&grade_str),
        score = score.total,
        period = xml_escape(&analysis.period_label),
        commits = analysis.total_commits,
        files = analysis.total_files,
        bars = bar_svgs,
        mood_label_y = 395,
        mood_bar = mood_bar,
    )
}

/// XML 특수문자 이스케이프
fn xml_escape(s: &str) -> String {
    s.replace('&', "&amp;")
     .replace('<', "&lt;")
     .replace('>', "&gt;")
     .replace('"', "&quot;")
}

/// PNG 변환 (resvg feature flag 필요)
#[cfg(feature = "png")]
pub fn save_as_png(svg_content: &str, path: &std::path::Path) -> Result<(), crate::error::VibeError> {
    use resvg::tiny_skia;
    use resvg::usvg;

    let opt = usvg::Options::default();
    let tree = usvg::Tree::from_str(svg_content, &opt)
        .map_err(|e| crate::error::VibeError::Git(format!("SVG 파싱 오류: {}", e)))?;

    let size = tree.size();
    let width = size.width() as u32;
    let height = size.height() as u32;

    let mut pixmap = tiny_skia::Pixmap::new(width, height)
        .ok_or_else(|| crate::error::VibeError::Git("PNG 버퍼 생성 실패".to_string()))?;

    resvg::render(&tree, tiny_skia::Transform::default(), &mut pixmap.as_mut());

    pixmap.save_png(path)
        .map_err(|e| crate::error::VibeError::Io(
            std::io::Error::new(std::io::ErrorKind::Other, e.to_string())
        ))
}

#[cfg(not(feature = "png"))]
pub fn save_as_png(_svg_content: &str, _path: &std::path::Path) -> Result<(), crate::error::VibeError> {
    Err(crate::error::VibeError::Git(
        "PNG 지원은 'png' feature가 필요합니다. `cargo install git-vibe --features png`로 설치하세요".to_string()
    ))
}
```

## 3. S-02: README 배지 SVG

### 3.1 설계

**신규 파일**: `src/output/badge.rs`

shields.io 호환 배지 SVG를 문자열로 생성합니다.

**배지 사양**:
- 높이: 20px (shields.io 표준)
- 왼쪽: "vibe" 라벨 (회색 배경)

```rust
/// shields.io 스타일 배지 SVG 생성
pub fn render_badge(score: &VibeScore) -> String {
    let grade_str = score.grade.to_string();
    let value_text = format!("{} ({:.0})", grade_str, score.total);
    let color = badge_color(&score.grade);

    // 텍스트 폭 추정 (문자당 약 6.5px, 이모지 약 12px)
    let label = "vibe";
    let label_width = 40u32;
    let value_width = (value_text.len() as u32 * 7) + 10;
    let total_width = label_width + value_width;

    format!(
        r#"<svg xmlns="http://www.w3.org/2000/svg" width="{total}" height="20" viewBox="0 0 {total} 20">
  <linearGradient id="s" x2="0" y2="100%">
    <stop offset="0" stop-color="#bbb" stop-opacity=".1"/>
    <stop offset="1" stop-opacity=".1"/>
  </linearGradient>
  <clipPath id="r"><rect width="{total}" height="20" rx="3" fill="#fff"/></clipPath>
  <g clip-path="url(#r)">
    <rect width="{lw}" height="20" fill="#555"/>
    <rect x="{lw}" width="{vw}" height="20" fill="{color}"/>
    <rect width="{total}" height="20" fill="url(#s)"/>
  </g>
  <g fill="#fff" text-anchor="middle" font-family="Verdana,Geneva,sans-serif" font-size="11">
    <text x="{lx}" y="15" fill="#010101" fill-opacity=".3">{label}</text>
    <text x="{lx}" y="14">{label}</text>
    <text x="{vx}" y="15" fill="#010101" fill-opacity=".3">{value}</text>
    <text x="{vx}" y="14">{value}</text>
  </g>
</svg>"#,
        total = total_width,
        lw = label_width,
        vw = value_width,
        color = color,
        lx = label_width / 2,
        vx = label_width + value_width / 2,
        label = label,
        value = xml_escape(&value_text),
    )
}

fn xml_escape(s: &str) -> String {
    s.replace('&', "&amp;")
     .replace('<', "&lt;")
     .replace('>', "&gt;")
     .replace('"', "&quot;")
}
```

### 4.4 entrypoint.sh

```bash
#!/bin/bash
set -e

PERIOD="${1:-6m}"
POST_COMMENT="${2:-true}"

# git-vibe 실행 (JSON 출력)
RESULT=$(git-vibe --path "${GITHUB_WORKSPACE}" --period "${PERIOD}" --json)

# JSON에서 값 추출
VIBE_SCORE=$(echo "$RESULT" | jq -r '.vibe_score')
VIBE_GRADE=$(echo "$RESULT" | jq -r '.vibe_grade')
BUS_FACTOR=$(echo "$RESULT" | jq -r '.bus_factor')

# 서브 스코어 추출
FREQ=$(echo "$RESULT" | jq -r '.sub_scores.commit_frequency')
HOTSPOT=$(echo "$RESULT" | jq -r '.sub_scores.hotspot_score')
GHOST=$(echo "$RESULT" | jq -r '.sub_scores.ghost_score')
MOOD=$(echo "$RESULT" | jq -r '.sub_scores.mood_score')
BF_SCORE=$(echo "$RESULT" | jq -r '.sub_scores.bus_factor_score')
TREND=$(echo "$RESULT" | jq -r '.sub_scores.trend_score')

# Markdown 코멘트 생성
COMMENT="## 🎭 Vibe Check

**Overall Vibe: ${VIBE_GRADE} (${VIBE_SCORE}/100)**

| Metric | Score |
|--------|------:|
| Commit Frequency | ${FREQ} |
| Hotspot Health | ${HOTSPOT} |
| Ghost Files | ${GHOST} |
| Commit Mood | ${MOOD} |
| Bus Factor | ${BF_SCORE} |
| Activity Trend | ${TREND} |

🏋️ Bus Factor: ${BUS_FACTOR}

---
*Generated by [git-vibe](https://github.com/calintzy/git-vibe)*"

# GitHub Output 설정
echo "vibe-score=${VIBE_SCORE}" >> "$GITHUB_OUTPUT"
echo "vibe-grade=${VIBE_GRADE}" >> "$GITHUB_OUTPUT"

# PR 코멘트 게시
if [ "$POST_COMMENT" = "true" ] && [ -n "$GITHUB_TOKEN" ]; then
    if [ -n "$GITHUB_EVENT_PATH" ]; then
        PR_NUMBER=$(jq -r '.pull_request.number // empty' "$GITHUB_EVENT_PATH")
        if [ -n "$PR_NUMBER" ]; then
            PAYLOAD=$(jq -n --arg body "$COMMENT" '{"body": $body}')
            curl -s -X POST \
                -H "Authorization: token ${GITHUB_TOKEN}" \
                -H "Accept: application/vnd.github.v3+json" \
                "${GITHUB_API_URL}/repos/${GITHUB_REPOSITORY}/issues/${PR_NUMBER}/comments" \
                -d "$PAYLOAD"
        fi
    fi
fi

echo "$COMMENT"
```

변경 사항:
- `version` → `"0.2.0"`

## 6. 구현 순서

```
┌─────────────────────────────────────────────────────────┐
│ Step 1: CLI 옵션 추가                                     │
│   --share, --format, -o, --badge 옵션                    │
│   → src/cli.rs 수정                                     │
├─────────────────────────────────────────────────────────┤
│ Step 2: SVG 렌더링 엔진                                   │
│   render_svg(), score_bar_svg(), grade_color()           │
│   → src/output/svg.rs 신규                              │
├─────────────────────────────────────────────────────────┤
│ Step 3: 배지 SVG 생성                                     │
│   render_badge(), badge_color()                          │
│   → src/output/badge.rs 신규                            │
├─────────────────────────────────────────────────────────┤
│ Step 4: output/mod.rs + main.rs 통합                      │
│   render_share(), render_badge() 추가                    │
│   main.rs 분기 추가                                      │
│   → src/output/mod.rs, src/main.rs 수정                 │
├─────────────────────────────────────────────────────────┤
│ Step 5: Cargo.toml + PNG feature flag                    │
│   version 0.2.0, [features] png, resvg optional         │
│   → Cargo.toml 수정                                     │
├─────────────────────────────────────────────────────────┤
│ Step 6: GitHub Action                                    │
│   action.yml, Dockerfile, entrypoint.sh                  │
│   → .github/actions/git-vibe/ 신규                      │
├─────────────────────────────────────────────────────────┤
│ Step 7: 테스트 + 검증                                     │
│   SVG 유효성, 배지 생성, CLI 통합 테스트                    │
│   → tests/cli_tests.rs 수정, 수동 검증                   │
└─────────────────────────────────────────────────────────┘
```

### 구현 순서 근거

| 순서 | 이유 |
|------|------|
| Step 1 (CLI) 최우선 | 다른 모든 기능의 진입점. 구조체 변경 먼저 확정 |
| Step 2 (SVG) 두 번째 | 핵심 기능. 배지와 PNG 모두 SVG에서 파생 |
| Step 3 (배지) 세 번째 | SVG 엔진과 유사한 패턴, 독립적 |
| Step 4 (통합) 네 번째 | Step 1-3 완료 후 조립 |
| Step 5 (Cargo) 다섯 번째 | feature flag 추가, 빌드 확인 |
| Step 6 (Action) 여섯 번째 | CLI 완성 후 래핑, 독립적 |
| Step 7 (테스트) 마지막 | 전체 검증 |

### 8.1 단위 테스트

### 8.3 수동 검증

```bash
# SVG 생성 후 브라우저에서 확인
git-vibe --share --path /path/to/repo
open vibe-report.svg

# 배지 생성
git-vibe --badge --path /path/to/repo
open vibe-badge.svg

# PNG 생성 (feature flag)
cargo run --features png -- --share --format png --path /path/to/repo

# JSON과 동시 사용 불가 확인
git-vibe --share --json  # 에러 또는 share 우선
```
